    ```python
    async def _handle_rvc_test_voice(self, payload):
        """Record 3 seconds from mic, convert through RVC, send back for playback."""
        rvc = self._tts.get_rvc() if self._tts else None
        if not rvc or not rvc.is_enabled():
            await self.broadcast(create_event('rvc_test_voice_error', {
                'error': 'RVC is not enabled or no model loaded'
            }))
            return

        try:
            # Notify frontend that recording is starting
            await self.broadcast(create_event('rvc_test_voice_recording', {
                'duration_seconds': 3
            }))

            # Record 3 seconds from the user's microphone
            import sounddevice as sd
            sample_rate = 16000  # Record at 16kHz for HuBERT
            duration = 3  # seconds
            channels = 1  # mono

            input_device = self.settings.get('audio', {}).get('input_device', None)

            # Run recording in executor to avoid blocking
            def record_sync():
                return sd.rec(
                    int(duration * sample_rate),
                    samplerate=sample_rate,
                    channels=channels,
                    dtype='float32',
                    device=input_device
                )

            recording = await asyncio.get_event_loop().run_in_executor(None, record_sync)

            # Wait for recording to finish
            await asyncio.sleep(duration + 0.5)  # Extra 0.5s buffer
            sd.wait()  # Ensure recording is complete

            # Flatten to 1D
            audio_data = recording.flatten()

            # Create a TTSResult from the recording (WAV format at 16kHz)
            import io, wave
            wav_buffer = io.BytesIO()
            with wave.open(wav_buffer, 'wb') as wf:
                wf.setnchannels(1)
                wf.setsampwidth(2)  # 16-bit
                wf.setframerate(sample_rate)
                # Convert float32 to int16
                int_data = (audio_data * 32767).astype(np.int16)
                wf.writeframes(int_data.tobytes())

            from ai.tts.base import TTSResult
            tts_result = TTSResult(
                audio_data=wav_buffer.getvalue(),
                sample_rate=sample_rate,
                channels=1,
                sample_width=2
            )

            # Process through RVC
            converted = await rvc.process(tts_result)

            # Encode as base64 WAV for frontend playback
            import base64
            audio_b64 = base64.b64encode(converted.audio_data).decode('utf-8')

            await self.broadcast(create_event('rvc_test_voice_ready', {
                'audio_base64': audio_b64,
                'sample_rate': converted.sample_rate
            }))

        except Exception as e:
            logger.error(f"Test voice failed: {e}")
            await self.broadcast(create_event('rvc_test_voice_error', {
                'error': str(e)
            }))
    ```
